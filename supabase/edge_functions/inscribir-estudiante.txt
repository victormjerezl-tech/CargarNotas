import { serve } from "https://deno.land/std@0.177.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

function getServiceRoleKey() {
  const secretKeys = Deno.env.get("SUPABASE_SECRET_KEYS");
  if (!secretKeys) {
    throw new Error("SUPABASE_SECRET_KEYS is required");
  }
  try {
    const parsed = JSON.parse(secretKeys);
    return parsed.service_role ?? parsed.serviceRole ?? parsed.service_role_key ?? parsed.serviceRoleKey ?? Object.values(parsed)[0] ?? secretKeys;
  } catch {
    return secretKeys;
  }
}

serve(async (req: Request) => {
  try {
    if (req.method !== "POST") {
      return new Response(
        JSON.stringify({ ok: false, error: "Método no permitido." }),
        { status: 405 }
      );
    }

    const { estudiante_id, seccion_id, anio_escolar_id } = await req.json();

    if (!estudiante_id || !seccion_id || !anio_escolar_id) {
      return new Response(
        JSON.stringify({ ok: false, error: "Faltan parámetros obligatorios." }),
        { status: 400 }
      );
    }

    const supabase = createClient(
      Deno.env.get("SUPABASE_URL")!,
      getServiceRoleKey()!,
      {
        global: {
          headers: { Authorization: req.headers.get("Authorization")! },
        },
      }
    );

    // 1. Verificar estudiante
    const { data: est, error: errEst } = await supabase
      .from("estudiantes")
      .select("id_estudiante, activo")
      .eq("id_estudiante", estudiante_id)
      .single();

    if (errEst || !est) {
      return new Response(
        JSON.stringify({ ok: false, error: "El estudiante no existe." }),
        { status: 400 }
      );
    }

    if (!est.activo) {
      return new Response(
        JSON.stringify({ ok: false, error: "El estudiante está inactivo." }),
        { status: 400 }
      );
    }

    // 2. Verificar sección
    const { data: sec, error: errSec } = await supabase
      .from("secciones")
      .select("id_seccion, activo")
      .eq("id_seccion", seccion_id)
      .single();

    if (errSec || !sec) {
      return new Response(
        JSON.stringify({ ok: false, error: "La sección no existe." }),
        { status: 400 }
      );
    }

    if (!sec.activo) {
      return new Response(
        JSON.stringify({ ok: false, error: "La sección está inactiva." }),
        { status: 400 }
      );
    }

    // 3. Verificar año escolar
    const { data: anio, error: errAnio } = await supabase
      .from("anios_escolares")
      .select("id_anio, activo")
      .eq("id_anio", anio_escolar_id)
      .single();

    if (errAnio || !anio) {
      return new Response(
        JSON.stringify({ ok: false, error: "El año escolar no existe." }),
        { status: 400 }
      );
    }

    if (!anio.activo) {
      return new Response(
        JSON.stringify({ ok: false, error: "El año escolar está inactivo." }),
        { status: 400 }
      );
    }

    // 4. Insertar inscripción
    const { data, error } = await supabase
      .from("inscripciones")
      .insert({
        estudiante_id,
        seccion_id,
        anio_escolar_id,
        estado: "activo",
      })
      .select()
      .single();

    if (error) {
      return new Response(
        JSON.stringify({ ok: false, error: error.message }),
        { status: 400 }
      );
    }

    return new Response(
      JSON.stringify({
        ok: true,
        mensaje: "Estudiante inscrito correctamente.",
        inscripcion: data,
      }),
      { status: 200 }
    );

  } catch (err) {
    return new Response(
      JSON.stringify({ ok: false, error: err.message }),
      { status: 500 }
    );
  }
});
