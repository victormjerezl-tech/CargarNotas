// listar-estudiantes/index.ts

import { serve } from "https://deno.land/std@0.177.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

function getServiceRoleKey() {
  const secretKeys = Deno.env.get("SUPABASE_SECRET_KEYS");
  if (!secretKeys) {
    throw new Error("SUPABASE_SECRET_KEYS is required");
  }
  try {
    const parsed = JSON.parse(secretKeys);
    return parsed.service_role ?? parsed.serviceRole ?? parsed.service_role_key ?? parsed.serviceRoleKey ?? Object.values(parsed)[0] ?? secretKeys;
  } catch {
    return secretKeys;
  }
}

serve(async (req: Request) => {
  try {
    if (req.method !== "POST") {
      return new Response(
        JSON.stringify({ ok: false, error: "Método no permitido." }),
        { status: 405 }
      );
    }

    const {
      buscar,
      anio_escolar_id,
      grado_id,
      seccion_id,
      estado,
      ordenar_por = "apellidos",
      direccion = "asc",
      pagina = 1,
      limite = 20
    } = await req.json();

    const supabase = createClient(
      Deno.env.get("SUPABASE_URL")!,
      getServiceRoleKey()!,
      {
        global: {
          headers: { Authorization: req.headers.get("Authorization")! },
        },
      }
    );

    // Construcción dinámica del query
    let query = supabase
      .from("estudiantes")
      .select(`
        id_estudiante,
        cedula,
        nombres,
        apellidos,
        fecha_nacimiento,
        sexo,
        activo,
        inscripciones (
          id_inscripcion,
          seccion_id,
          anio_escolar_id,
          estado
        )
      `, { count: "exact" });

    // Filtro: búsqueda
    if (buscar && buscar.trim() !== "") {
      query = query.or(
        `cedula.ilike.%${buscar}%,nombres.ilike.%${buscar}%,apellidos.ilike.%${buscar}%`
      );
    }

    // Filtro: año escolar
    if (anio_escolar_id) {
      query = query.contains("inscripciones", [{ anio_escolar_id }]);
    }

    // Filtro: grado
    if (grado_id) {
      query = query.contains("inscripciones", [{ grado_id }]);
    }

    // Filtro: sección
    if (seccion_id) {
      query = query.contains("inscripciones", [{ seccion_id }]);
    }

    // Filtro: estado del estudiante
    if (estado === "activo") query = query.eq("activo", true);
    if (estado === "inactivo") query = query.eq("activo", false);

    // Ordenamiento
    query = query.order(ordenar_por, { ascending: direccion === "asc" });

    // Paginación
    const desde = (pagina - 1) * limite;
    const hasta = desde + limite - 1;

    query = query.range(desde, hasta);

    // Ejecutar query
    const { data, error, count } = await query;

    if (error) {
      return new Response(
        JSON.stringify({ ok: false, error: error.message }),
        { status: 400 }
      );
    }

    return new Response(
      JSON.stringify({
        ok: true,
        total: count,
        pagina,
        limite,
        estudiantes: data
      }),
      { status: 200 }
    );

  } catch (err) {
    return new Response(
      JSON.stringify({ ok: false, error: err.message }),
      { status: 500 }
    );
  }
});
