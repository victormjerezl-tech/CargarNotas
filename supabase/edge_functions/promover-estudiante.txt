// promover-estudiante/index.ts

import { serve } from "https://deno.land/std@0.177.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

function getServiceRoleKey() {
  const secretKeys = Deno.env.get("SUPABASE_SECRET_KEYS");
  if (!secretKeys) {
    throw new Error("SUPABASE_SECRET_KEYS is required");
  }
  try {
    const parsed = JSON.parse(secretKeys);
    return parsed.service_role ?? parsed.serviceRole ?? parsed.service_role_key ?? parsed.serviceRoleKey ?? Object.values(parsed)[0] ?? secretKeys;
  } catch {
    return secretKeys;
  }
}

serve(async (req: Request) => {
  try {
    if (req.method !== "POST") {
      return new Response(
        JSON.stringify({ ok: false, error: "Método no permitido." }),
        { status: 405 }
      );
    }

    const { id_inscripcion, nuevo_grado, nueva_seccion } = await req.json();

    if (!id_inscripcion || !nuevo_grado || !nueva_seccion) {
      return new Response(
        JSON.stringify({ ok: false, error: "Faltan parámetros obligatorios." }),
        { status: 400 }
      );
    }

    const supabase = createClient(
      Deno.env.get("SUPABASE_URL")!,
      getServiceRoleKey()!,
      {
        global: {
          headers: { Authorization: req.headers.get("Authorization")! },
        },
      }
    );

    // 1. Obtener inscripción actual
    const { data: insc, error: errInsc } = await supabase
      .from("inscripciones")
      .select("*")
      .eq("id_inscripcion", id_inscripcion)
      .single();

    if (errInsc || !insc) {
      return new Response(
        JSON.stringify({ ok: false, error: "La inscripción no existe." }),
        { status: 400 }
      );
    }

    if (insc.estado !== "activo") {
      return new Response(
        JSON.stringify({ ok: false, error: "Solo se pueden promover inscripciones activas." }),
        { status: 400 }
      );
    }

    // 2. Marcar inscripción actual como promovida
    const { error: errUpdate } = await supabase
      .from("inscripciones")
      .update({ estado: "promovido" })
      .eq("id_inscripcion", id_inscripcion);

    if (errUpdate) {
      return new Response(
        JSON.stringify({ ok: false, error: errUpdate.message }),
        { status: 400 }
      );
    }

    // 3. Crear nueva inscripción
    const { data: nueva, error: errNueva } = await supabase
      .from("inscripciones")
      .insert({
        estudiante_id: insc.estudiante_id,
        seccion_id: nueva_seccion,
        anio_escolar_id: insc.anio_escolar_id,
        estado: "activo"
      })
      .select()
      .single();

    if (errNueva) {
      return new Response(
        JSON.stringify({ ok: false, error: errNueva.message }),
        { status: 400 }
      );
    }

    return new Response(
      JSON.stringify({
        ok: true,
        mensaje: "Estudiante promovido correctamente.",
        id_estudiante: insc.estudiante_id,
        nueva_inscripcion: nueva
      }),
      { status: 200 }
    );

  } catch (err) {
    return new Response(
      JSON.stringify({ ok: false, error: err.message }),
      { status: 500 }
    );
  }
});
