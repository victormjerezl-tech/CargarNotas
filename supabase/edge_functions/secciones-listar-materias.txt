// secciones-listar-materias/index.ts
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

function getServiceRoleKey() {
  const secretKeys = Deno.env.get("SUPABASE_SECRET_KEYS");
  if (!secretKeys) {
    throw new Error("SUPABASE_SECRET_KEYS is required");
  }
  try {
    const parsed = JSON.parse(secretKeys);
    return parsed.service_role ?? parsed.serviceRole ?? parsed.service_role_key ?? parsed.serviceRoleKey ?? Object.values(parsed)[0] ?? secretKeys;
  } catch {
    return secretKeys;
  }
}

const supabaseAdmin = createClient(
  Deno.env.get("SUPABASE_URL")!,
  getServiceRoleKey()!
);

Deno.serve(async (req) => {
  try {
    if (req.method !== "GET") {
      return new Response(
        JSON.stringify({ error: "Método no permitido" }),
        { status: 405 }
      );
    }

    // 1. Validar token
    const authHeader = req.headers.get("Authorization") ?? "";
    const token = authHeader.replace("Bearer ", "");
    if (!token) {
      return new Response(JSON.stringify({ error: "Token requerido" }), {
        status: 401,
      });
    }

    const { data: authData, error: authError } = await supabaseAdmin.auth.getUser(token);
    if (authError || !authData.user) {
      return new Response(JSON.stringify({ error: "Token inválido" }), {
        status: 401,
      });
    }

    const user_id = authData.user.id;

    // 2. Validar roles permitidos

    const { data: roles } = await supabaseAdmin
      .from("user_roles")
      .select("id_rol")
      .eq("user_id", user_id);

    // Roles permitidos:
    // 1 = superadmin
    // 2 = directivo
    // 3 = evaluacion_docente (puede ver)
    // 4 = control_estudios
    // 1 = docente (puede ver sus materias)
    const rolesPermitidos = [1, 2, 3, 4];
    const tienePermiso = roles?.some((r) =>
      rolesPermitidos.includes(r.id_rol)
    );

    if (!tienePermiso) {
      return new Response(
        JSON.stringify({ error: "No tiene permisos para ver materias de la sección" }),
        { status: 403 }
      );
    }

    // 3. Obtener parámetros de querystring
    const url = new URL(req.url);
    const id_seccion = url.searchParams.get("id_seccion");

    if (!id_seccion) {
      return new Response(
        JSON.stringify({ error: "id_seccion es obligatorio" }),
        { status: 400 }
      );
    }

    // 4. Llamar a la función SQL
    const { data, error } = await supabaseAdmin.rpc("listar_materias_seccion", {
      p_id_seccion: Number(id_seccion),
    });

    if (error) {
      return new Response(JSON.stringify({ error: error.message }), {
        status: 400,
      });
    }

    return new Response(JSON.stringify(data), { status: 200 });

  } catch (err) {
    return new Response(JSON.stringify({ error: err.message }), {
      status: 500,
    });
  }
});
